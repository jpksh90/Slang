 fun apply_n(f, n, x) => if (n == 0.0) then x else f(apply_n(f, n - 1.0, f(x)));
 fun apply(f, x) => f(x);
 fun fixpoint(f, x) => if (f(x) == x) then x else fixpoint(f, f(x));
 fun double(f, x) => f(f(x));
 fun test() {
  {
     let f =  fun anonymous~0(x) => x + 1.0;
     let x = 0.0;
     let n = 10.0;
     let y = apply_n(f, n, x);
     let z = apply(f, x);
     let w = fixpoint(f, x);
     let v = double(f, x);
     print(y);
     print(z);
     print(w);
     print(v);
  }
};
