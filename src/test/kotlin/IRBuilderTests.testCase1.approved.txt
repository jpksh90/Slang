 fun __module__main__() {
  {
     let base = readInput();
     let exp = readInput();
     print(power(base, exp));
  }
}
 fun power(base, exp) {
  {
     let result = 1.0;
     while (exp > 0.0) {
      {
        result = result * base;
        exp = exp - 1.0;
      }
    }
     return result;
  }
}
