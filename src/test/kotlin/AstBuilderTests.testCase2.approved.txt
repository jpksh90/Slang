=>IfExpr(condition=BinaryExpr(left=VarExpr(name=n), op===, right=IntExpr(value=0)), thenExpr=IntExpr(value=1), elseExpr=BinaryExpr(left=VarExpr(name=n), op=*, right=FuncCallExpr(target=factorial, args=[BinaryExpr(left=VarExpr(name=n), op=-, right=IntExpr(value=1))])))
